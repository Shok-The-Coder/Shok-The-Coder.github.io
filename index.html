Kill Brick Code

The most basic kind of code out there!

-- Server Script, make sure to put inside a part--

local killBrick = script.Parent

local function touch(object)
local character = object.Parent
local humanoid = character:FindFirstChild("Humanoid")
if humanoid then
humanoid.Health = 0
end
end

killBrick.Touched:Connect(touch)

Healing Brick Code

Also a very basic code that many developers use.

-- Server Script, please put inside a part --

local healBrick = script.Parent

local function touch(object)
local character = object.Parent
local humanoid = character:FindFirstChild("Humanoid")
if humanoid then
humanoid.Health += 1 -- Generates 1 hp everytime(don't need to type this, just a comment, like a reminder can be helpful)
end
end
healBrick.Touched:Connect(touch)

Click Detector Dialogue

-- Server Script, add a click detector to the part and a server script, and a model called characterParts --

local Chat = game:GetService("Chat")

local ncp = script.Parent
local characterParts = ncp.CharacterParts
local head = characterParts.Head
local clickDetector = ncp.ClickDetector

local dialogues = {"Hello!", "How are you?", "Good bye!"}
local dialogueNumber = 1

local function talk()
local dialogue = dialogues[dialogueNumber]
Chat:Chat(head, dialogue)
if dialogueNumber == #dialogues then
dialogueNumber = 1
else
dialogueNumber += 1
end
end

clickDetector.MouseClick:Connect(talk)
